## FASE 2 — Frontend: Config (`frontend/src/lib/config.ts`)

Crear si no existe. Si existe, verificar que tiene este contenido exacto:
```typescript
const API_BASE_URL = import.meta.env.PUBLIC_API_URL;

export async function fetchApi<T>(
  endpoint: string,
  options?: RequestInit
): Promise<T> {
  const url = `${API_BASE_URL}${endpoint}`;

  const config: RequestInit = {
    ...options,
    headers: { ...options?.headers },
  };

  if (!(options?.body instanceof FormData)) {
    config.headers = {
      ...config.headers,
      "Content-Type": "application/json",
    };
  }

  const response = await fetch(url, config);

  if (!response.ok) {
    let errorMessage = "Error desconocido del servidor";
    try {
      const errorData = await response.json();
      errorMessage = errorData.detail || errorMessage;
    } catch {
      // usar mensaje por defecto
    }
    throw new Error(errorMessage);
  }

  return response.json() as Promise<T>;
}
```

Agregar al `frontend/.env` si no está:

Patrón para sección única (hero, about — PUT sobre el registro existente):
```svelte
<!-- frontend/src/components/admin/HeroEditor.svelte -->
<script lang="ts">
  import { onMount } from 'svelte';
  import { fetchApi } from '$lib/config';
  import type { HeroResponse } from '$lib/types';

  const API = import.meta.env.PUBLIC_API_URL;

  let data: HeroResponse | null = null;
  let loading = true;
  let saving = false;
  let message = '';
  let messageType: 'success' | 'error' = 'success';

  // un campo por cada campo del schema Create
  let title = '';
  let subtitle = '';
  let imageFile: File | null = null;

  onMount(async () => {
    try {
      data = await fetchApi<HeroResponse>('/api/v1/heroes/latest/');
      title    = data.title    ?? '';
      subtitle = data.subtitle ?? '';
    } catch {
      message     = 'Error al cargar los datos';
      messageType = 'error';
    } finally {
      loading = false;
    }
  });

  async function handleSubmit() {
    if (!data) return;
    saving  = true;
    message = '';

    try {
      // SIEMPRE FormData — nunca JSON
      const formData = new FormData();
      formData.append('title', title);
      formData.append('subtitle', subtitle);
      if (imageFile) formData.append('image', imageFile);
      // NO agregar Content-Type manualmente

      const res = await fetch(`${API}/api/v1/heroes/${data.id}`, {
        method: 'PUT',
        body: formData,
      });

      if (!res.ok) throw new Error();
      data        = await res.json();
      message     = 'Guardado correctamente';
      messageType = 'success';
    } catch {
      message     = 'Error al guardar los cambios';
      messageType = 'error';
    } finally {
      saving = false;
    }
  }
</script>

{#if loading}
  <div class="flex items-center gap-2 text-zinc-400">
    <span class="animate-spin inline-block w-4 h-4 border-2
                 border-zinc-600 border-t-amber-400 rounded-full"></span>
    <span class="text-sm">Cargando...</span>
  </div>
{:else}
  <form on:submit|preventDefault={handleSubmit} class="space-y-6">

    <!-- Imagen actual + upload -->
    <div>
      <label class="block text-sm font-medium text-zinc-300 mb-2">
        Imagen
      </label>
      {#if data?.image_url}
        <img
          src={data.image_url}
          alt="Imagen actual"
          class="w-32 h-32 object-cover rounded-lg mb-3 border border-zinc-700"
        />
      {/if}
      <input
        type="file"
        accept="image/*"
        on:change={(e) => {
          const input = e.target as HTMLInputElement;
          imageFile = input.files?.[0] ?? null;
        }}
        class="block w-full text-sm text-zinc-400
               file:mr-4 file:py-2 file:px-4
               file:rounded-lg file:border-0
               file:bg-zinc-800 file:text-amber-400
               hover:file:bg-zinc-700 cursor-pointer"
      />
      <p class="text-xs text-zinc-500 mt-1">
        Deja vacío para mantener la imagen actual
      </p>
    </div>

    <!-- Campo texto — repetir por cada campo del schema -->
    <div>
      <label class="block text-sm font-medium text-zinc-300 mb-2">
        Título
      </label>
      <input
        type="text"
        bind:value={title}
        required
        class="w-full px-4 py-2.5 rounded-lg bg-zinc-800 border border-zinc-700
               text-zinc-100 placeholder-zinc-500
               focus:outline-none focus:border-amber-400 focus:ring-1
               focus:ring-amber-400 transition-colors"
        placeholder="Título principal"
      />
    </div>

    <div>
      <label class="block text-sm font-medium text-zinc-300 mb-2">
        Subtítulo
      </label>
      <textarea
        bind:value={subtitle}
        rows={4}
        class="w-full px-4 py-2.5 rounded-lg bg-zinc-800 border border-zinc-700

               focus:outline-none focus:border-amber-400 focus:ring-1
               focus:ring-amber-400 transition-colors"
        placeholder="Subtítulo o descripción"
      ></textarea>
    </div>

    <!-- Botón guardar + mensaje -->
    <div class="flex items-center gap-4">
      <button
        type="submit"
        disabled={saving}
        class="px-6 py-2.5 rounded-lg font-medium text-sm
               bg-gradient-to-r from-amber-400 to-orange-500
               text-zinc-900 hover:from-amber-300 hover:to-orange-400
               disabled:opacity-50 disabled:cursor-not-allowed
               transition-all duration-200"
      >
        {saving ? 'Guardando...' : 'Guardar cambios'}
      </button>

      {#if message}
        <span
          class="text-sm {messageType === 'success'
            ? 'text-emerald-400'
            : 'text-red-400'}"
        >
          {message}
        </span>
      {/if}
    </div>
  </form>
{/if}
```

## Reglas específicas para archivos `.astro`

- ❌ Nunca usar `.map()` con arrow functions directamente en el template
  HTML de un archivo `.astro` — el parser de Astro no es JSX y falla
  con expresiones complejas dentro de `{}`
- ✅ En su lugar, preparar los datos en el frontmatter (`---`) y usar
  `{#each}` de Svelte en componentes `.svelte`, o iterar en Astro así:
```astro

  // preparar datos en el frontmatter
  const sections = [
    { href: '/admin/hero', label: 'Hero', description: '...' },

  <!-- iterar con for...of en Astro, NO con .map() -->

  ))}
```

  Svelte con `{#each}` y cargarlo con `client:load`.

- ❌ Nunca importar componentes Lucide directamente en archivos `.astro`
  para renderizarlos — Lucide Svelte solo funciona dentro de
  componentes `.svelte`
- ✅ Para iconos en páginas `.astro` usar SVG inline o mover el bloque

- ❌ Nunca usar expresiones ternarias complejas o encadenamiento de

## Reglas específicas para Svelte 5

- ❌ Nunca poner un bloque `{#if}` o `{#each}` que abra dentro de un
  elemento HTML y cierre fuera — Svelte 5 es estricto con el anidamiento
- ❌ Nunca colocar un tooltip como hijo directo de `<a>` con un `{#if}`
  que cruce el cierre del `<a>`
- ✅ Para tooltips en modo colapsado: envolver el `<a>` y el tooltip en
  un `<div class="relative group">` — el tooltip va FUERA del `<a>`
  como elemento hermano, nunca dentro
- ✅ Los bloques `{#if}` dentro de elementos deben abrirse y cerrarse
  completamente dentro de ese elemento, sin cruzar su tag de cierre
- ❌ Nunca usar `<svelte:component>` con `this={undefined}` — verificar
  siempre que el componente existe antes de renderizarlo
